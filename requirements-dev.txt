# JellyRip development dependencies
# Install with: pip install -r requirements-dev.txt
#
# These are the dependencies needed to run the test suite and
# build the bundle, in addition to the runtime requirements in
# requirements.txt.

# ===== Build =====
# PyInstaller bundles the .exe.  Pinned at >=6 because Phase 3f's
# spec uses the windowed-mode flag introduced in 6.x.
pyinstaller>=6.0

# ===== Runtime that the tests + bundle need =====
# PySide6 is the GUI toolkit for the Qt path (Phase 3 migration).
# Tkinter is bundled with Python on Windows and not pinned here.
PySide6>=6.5

# ===== Test framework =====
# pytest is the test runner.  Behavior-first tests (decision #5
# in docs/pyside6-migration-plan.md) make up the majority of the
# suite.
pytest>=7.0

# pytest-qt is required for any test that constructs a Qt widget.
# Phase 3a-3e shipped extensive Qt UI; tests that exercise widgets
# use pytest-qt's qtbot fixture.  Tests that don't need widgets
# (pure helpers, file-content introspection, etc.) skip pytest-qt
# via ``pytest.importorskip("pytestqt")`` at module level so they
# still run on environments without it.
pytest-qt>=4.0
